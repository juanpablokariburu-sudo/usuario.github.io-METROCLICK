    <script>

        // --- MOTOR DE ESTADO Y AUDIO ---

        let audioCtx;

        let isPlaying = false;

        

        let tempo = 80;

        let numerator = 4;

        let denominator = 4;

        let subdivisions = []; // Ej: [1, 2, 4, 3] = Negra, Corchea, Semicorchea, Tresillo

        

        // Rampa

        let isRamping = false;

        let rampStartBpm = 60;

        let rampEndBpm = 120;

        let rampDuration = 60;

        let rampStartTime = 0;


        // Temporización

        let currentBeat = 0;

        let currentSubPulse = 0;

        let nextNoteTime = 0.0;

        let lookahead = 25.0; // ms

        let scheduleAheadTime = 0.1; // sec

        let timerID;

        let notesQueue = [];


        // --- REFERENCIAS DOM ---

        const bpmInput = document.getElementById('bpm-input');

        const bpmSlider = document.getElementById('bpm-slider');

        const playBtn = document.getElementById('play-btn');

        const numInput = document.getElementById('num-input');

        const denInput = document.getElementById('den-input');

        const oscTypeSelect = document.getElementById('osc-type');

        const beatsContainer = document.getElementById('beats-container');

        const resetSubsBtn = document.getElementById('reset-subs');

        

        const rampStartInput = document.getElementById('ramp-start');

        const rampEndInput = document.getElementById('ramp-end');

        const rampTimeInput = document.getElementById('ramp-time');

        const startRampBtn = document.getElementById('start-ramp-btn');


        // --- INTERFAZ DINÁMICA DE PULSOS ---

        function renderBeats() {

            beatsContainer.innerHTML = '';

            

            // Mantener subdivisiones anteriores o llenar con 1

            while (subdivisions.length < numerator) subdivisions.push(1);

            subdivisions = subdivisions.slice(0, numerator);


            for (let i = 0; i < numerator; i++) {

                const beatCard = document.createElement('div');

                beatCard.id = `beat-card-${i}`;

                beatCard.className = 'beat-card bg-gray-800 border border-gray-600 rounded-xl p-3 flex flex-col items-center justify-between min-h-[140px]';

                

                const selectHtml = `

                    <div class="text-center w-full mb-3">

                        <span class="block text-[10px] text-gray-400 mb-1 uppercase tracking-widest">Tiempo ${i + 1}</span>

                        <select data-beat="${i}" class="sub-selector w-full bg-gray-900 text-yellow-400 font-bold border border-gray-700 rounded p-1.5 text-center focus:border-yellow-500 focus:outline-none appearance-none cursor-pointer">

                            <option value="1" ${subdivisions[i] == 1 ? 'selected' : ''}>1 (Negra)</option>

                            <option value="2" ${subdivisions[i] == 2 ? 'selected' : ''}>2 (Corcheas)</option>

                            <option value="3" ${subdivisions[i] == 3 ? 'selected' : ''}>3 (Tresillo)</option>

                            <option value="4" ${subdivisions[i] == 4 ? 'selected' : ''}>4 (Semicorcheas)</option>

                            <option value="5" ${subdivisions[i] == 5 ? 'selected' : ''}>5 (Quintillo)</option>

                            <option value="6" ${subdivisions[i] == 6 ? 'selected' : ''}>6 (Seisillo)</option>

                            <option value="7" ${subdivisions[i] == 7 ? 'selected' : ''}>7 (Septillo)</option>

                            <option value="9" ${subdivisions[i] == 9 ? 'selected' : ''}>9 (Nonillo)</option>

                        </select>

                    </div>

                `;

                

                let dotsHtml = `<div class="flex flex-wrap justify-center gap-1.5 w-full" id="dots-${i}">`;

                for (let j = 0; j < subdivisions[i]; j++) {

                    dotsHtml += `<div id="dot-${i}-${j}" class="sub-dot w-3 h-3 bg-gray-600 rounded-full"></div>`;

                }

                dotsHtml += `</div>`;


                beatCard.innerHTML = selectHtml + dotsHtml;

                beatsContainer.appendChild(beatCard);

            }


            document.querySelectorAll('.sub-selector').forEach(sel => {

                sel.addEventListener('change', (e) => {

                    const idx = parseInt(e.target.getAttribute('data-beat'));

                    subdivisions[idx] = parseInt(e.target.value);

                    updateDots(idx);

                });

            });

        }


        function updateDots(idx) {

            const container = document.getElementById(`dots-${idx}`);

            let html = '';

            for (let j = 0; j < subdivisions[idx]; j++) {

                html += `<div id="dot-${idx}-${j}" class="sub-dot w-3 h-3 bg-gray-600 rounded-full"></div>`;

            }

            container.innerHTML = html;

        }


        // --- LÓGICA DE AUDIO (SCHEDULING PRECIO) ---

        function nextNote() {

            // Manejo de la rampa de velocidad

            if (isRamping) {

                let elapsed = audioCtx.currentTime - rampStartTime;

                if (elapsed >= rampDuration) {

                    tempo = rampEndBpm;

                    isRamping = false;

                    startRampBtn.innerText = "Activar Ascenso/Descenso";

                    startRampBtn.classList.remove('bg-yellow-500', 'text-gray-900', 'animate-pulse');

                } else {

                    tempo = rampStartBpm + ((rampEndBpm - rampStartBpm) * (elapsed / rampDuration));

                }

                bpmInput.value = Math.round(tempo);

                bpmSlider.value = Math.round(tempo);

            }


            // Cálculo de duración real del pulso basado en el denominador

            // BPM tradicional es asumiendo negras (denominador 4).

            // Si el denominador es 8 (corchea), un compás es más rápido relativo al tempo.

            let beatRatio = 4 / denominator; 

            const secondsPerBeat = (60.0 / tempo) * beatRatio;

            

            const currentSubCount = subdivisions[currentBeat];

            const secondsPerSubPulse = secondsPerBeat / currentSubCount;


            nextNoteTime += secondsPerSubPulse;

            currentSubPulse++;

            

            if (currentSubPulse >= currentSubCount) {

                currentSubPulse = 0;

                currentBeat++;

                if (currentBeat >= numerator) currentBeat = 0;

            }

        }


        function scheduleNote(beatNum, subNum, time) {

            notesQueue.push({ beat: beatNum, sub: subNum, time: time });


            const osc = audioCtx.createOscillator();

            const env = audioCtx.createGain();


            osc.type = oscTypeSelect.value;

            

            // Frecuencias: Primer tiempo muy agudo, tiempos fuertes agudos, subdivisiones graves

            if (subNum === 0) {

                osc.frequency.value = (beatNum === 0) ? 1400 : 900;

            } else {

                osc.frequency.value = 450;

            }


            // Envolvente para evitar clicks de audio y hacer el sonido percusivo

            env.gain.value = 0.0;

            env.gain.linearRampToValueAtTime(1.0, time + 0.005);

            // Si hay muchas subdivisiones, el sonido debe apagarse más rápido

            let decay = (subdivisions[beatNum] > 4) ? 0.03 : 0.08;

            env.gain.exponentialRampToValueAtTime(0.001, time + decay);


            osc.connect(env);

            env.connect(audioCtx.destination);

            

            osc.start(time);

            osc.stop(time + decay + 0.01);

        }


        function scheduler() {

            while (nextNoteTime < audioCtx.currentTime + scheduleAheadTime) {

                scheduleNote(currentBeat, currentSubPulse, nextNoteTime);

                nextNote();

            }

            timerID = setTimeout(scheduler, lookahead);

        }


        // --- RENDERIZADO VISUAL ---

        function drawLoop() {

            if (isPlaying) {

                let currentAudioTime = audioCtx.currentTime;


                while (notesQueue.length && notesQueue[0].time < currentAudioTime) {

                    let note = notesQueue.shift();

                    

                    // Resetear estilos

                    document.querySelectorAll('.beat-card').forEach(el => el.classList.remove('beat-active'));

                    document.querySelectorAll('.sub-dot').forEach(el => el.className = 'sub-dot w-3 h-3 bg-gray-600 rounded-full');


                    // Activar panel

                    const card = document.getElementById(`beat-card-${note.beat}`);

                    if (card) card.classList.add('beat-active');


                    // Activar punto

                    const dot = document.getElementById(`dot-${note.beat}-${note.sub}`);

                    if (dot) {

                        if (note.sub === 0 && note.beat === 0) {

                            dot.classList.add('dot-measure-start');

                        } else if (note.sub === 0) {

                            dot.classList.add('dot-beat-start');

                        } else {

                            dot.classList.add('dot-subdivision');

                        }

                    }

                }

            } else {

                document.querySelectorAll('.beat-card').forEach(el => el.classList.remove('beat-active'));

                document.querySelectorAll('.sub-dot').forEach(el => el.className = 'sub-dot w-3 h-3 bg-gray-600 rounded-full');

            }

            requestAnimationFrame(drawLoop);

        }


        // --- EVENTOS DE INTERFAZ ---

        function updateTempo(newTempo) {

            tempo = Math.max(30, Math.min(300, newTempo));

            bpmInput.value = tempo;

            bpmSlider.value = tempo;

            if(isRamping) cancelRamp();

        }


        bpmInput.addEventListener('input', (e) => updateTempo(parseInt(e.target.value) || 80));

        bpmSlider.addEventListener('input', (e) => updateTempo(parseInt(e.target.value)));


        numInput.addEventListener('change', (e) => {

            numerator = parseInt(e.target.value) || 4;

            renderBeats();

            if(isPlaying && currentBeat >= numerator) currentBeat = 0;

        });


        denInput.addEventListener('change', (e) => {

            denominator = parseInt(e.target.value);

        });


        resetSubsBtn.addEventListener('click', () => {

            subdivisions.fill(1);

            renderBeats();

        });


        // Evento Play

        playBtn.addEventListener('click', () => {

            if (!audioCtx) {

                audioCtx = new (window.AudioContext || window.webkitAudioContext)();

                requestAnimationFrame(drawLoop);

            }

            if (audioCtx.state === 'suspended') audioCtx.resume();


            isPlaying = !isPlaying;


            if (isPlaying) {

                currentBeat = 0;

                currentSubPulse = 0;

                nextNoteTime = audioCtx.currentTime + 0.05;

                scheduler();

                

                playBtn.innerText = "DETENER";

        function cancelRamp() {

            isRamping = false;

            startRampBtn.innerText = "Activar Ascenso/Descenso";

            startRampBtn.classList.remove('bg-yellow-500', 'text-gray-900', 'animate-pulse');

        }

            if (!audioCtx) {

                audioCtx = new (window.AudioContext || window.webkitAudioContext)();

                requestAnimationFrame(drawLoop);

            }

            if (audioCtx.state === 'suspended') audioCtx.resume();


            if (isRamping) {

                cancelRamp();

                return;

            }


            rampStartBpm = parseFloat(rampStartInput.value);

            rampEndBpm = parseFloat(rampEndInput.value);

            rampDuration = parseFloat(rampTimeInput.value);


            if (rampDuration <= 0) return alert("El tiempo de rampa debe ser mayor a 0");


            updateTempo(rampStartBpm);

            isRamping = true;

            rampStartTime = audioCtx.currentTime;


            startRampBtn.innerText = "Rampa en progreso (Click para cancelar)";

            startRampBtn.classList.add('bg-yellow-500', 'text-gray-900', 'animate-pulse');


            if (!isPlaying) playBtn.click();

        });
